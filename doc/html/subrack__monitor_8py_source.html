57 class KeyReader :
58  '''
59  Read keypresses one at a time, without waiting for a newline.
60  echo: should characters be echoed?
61  block: should we block for each character, or return immediately?
62  (If !block, we'll return None if nothing is available to read.)

64  def __init__(self, echo=False, block=True):
65  '''Put the terminal into cbreak and noecho mode.'''
66  self.fd = sys.stdin.fileno()
67 
68  self.block = block
69 
70  self.oldterm = termios.tcgetattr(self.fd)
71  self.oldflags = fcntl.fcntl(self.fd, fcntl.F_GETFL)
72 
73  # Sad hack: when the destructor __del__ is called,
74  # the fcntl module may already be unloaded, so we can no longer
75  # call fcntl.fcntl() to set the terminal back to normal.
76  # So just in case, store a reference to the fcntl module,
77  # and also to termios (though I haven't yet seen a case
78  # where termios was gone -- for some reason it's just fnctl).
79  # The idea of keeping references to the modules comes from
80  # http://bugs.python.org/issue5099
81  # though I don't know if it'll solve the problem completely.
82  self.fcntl = fcntl
83  self.termios = termios
84 
85  newattr = termios.tcgetattr(self.fd)
86  # tcgetattr returns: [iflag, oflag, cflag, lflag, ispeed, ospeed, cc]
87  # where cc is a list of the tty special characters (length-1 strings)
88  # except for cc[termios.VMIN] and cc[termios.VTIME] which are ints.
89  self.cc_save = newattr[6]
90  newattr[3] = newattr[3] & ~termios.ICANON
91  if not echo:
92  newattr[3] = newattr[3] & ~termios.ECHO
93 
94  if block and False:
95  # VMIN and VTIME are supposed to let us do blocking reads:
96  # VMIN is the minimum number of characters before it will return,
97  # VTIME is how long it will wait if for characters < VMIN.
98  # This is documented in man termios.
99  # However, it doesn't work in python!
100  # In Python, read() never returns in non-canonical mode;
101  # even typing a newline doesn't help.
102  cc = self.cc_save[:] # Make a copy so we can restore VMIN, VTIME
103  cc[termios.VMIN] = 1
104  cc[termios.VTIME] = 0
105  newattr[6] = cc
106  else:
107  # Put stdin into non-blocking mode.
108  # We need to do this even if we're blocking, see above.
109  fcntl.fcntl(self.fd, fcntl.F_SETFL, self.oldflags | os.O_NONBLOCK)
110 
111  termios.tcsetattr(self.fd, termios.TCSANOW, newattr)
